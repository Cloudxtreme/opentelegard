inputform (length)

Displays a colored input bar of specified length and returns a value. This may sound simple, but in order to provide more precise input control, the input is read character by character into an array. The array is a set of numbers which represent the character code. Before returning the value, it is converted to a string. (replaces ansigets)

190:     def Input::inputform(length)
191:       trap("INT") do
192:         # ignore "CTRL-C"
193:       end
194:       tr = JLine::ConsoleReader.new
195:       char = nil
196:       enterkey = false
197:       input = []
198:       cursize = 0
199:       print ANSI_RESET+(ANSI_BRIGHT_WHITE+ANSI_ON_BLUE+" "*length)
200:       print "\b" * length
201:       # initialize
202:       # Start loop until enter key is pressed
203:       until enterkey == true
204:         #print ANSI_RESET
205:         char = tr.readVirtualKey
206:         # handle [ENTER] key
207:         if char == 10
208:           enterkey = true
209:         else
210:           # handle [Backspace] key
211:           if char == 8
212:             unless cursize == 0
213:               print "\b"+ANSI_BRIGHT_WHITE+ANSI_ON_BLUE+" \b"
214:               input.pop                        #remove last type value
215:               cursize -=1 unless cursize <= 0  #decrement cursize counter
216:             end
217:           # handle everything else
218:           else
219:             if cursize < length
220:               print ""+ANSI_BRIGHT_WHITE+ANSI_ON_BLUE+char.chr
221:               input.push(char)                  #add new value
222:               cursize += 1                      #increment cursize counter
223:             else
224:               # if the size of the current input is > length, then drop the char
225:               print "\b \b"
226:               char = nil
227:             end
228:           end
229:         end
230:         print ANSI_RESET
231:       end
232:       # convert input array of character codes back into string
233:       print ANSI_ON_BLUE+(" "*input.size)+ANSI_RESET+"\n"
234:       value = ""
235:       input.each do |c|
236:         value.concat(c)
237:       end
238:       return value
239:     end

inputkey ()

438: def Input::menuprompt(template, validkeys, menudata=nil)
439:   trap("INT") do
440:   end
441:   # Append globalkeys to supplied validkeys array. This is for DRY within each menu prompt.
442:   globalkeys = ['/?', '/G', '/H','/M','/P', '/T', '/W']
443:   validkeys = validkeys.concat(globalkeys)
444:   # Initialize loop condition
445:   done = false
446:   until done == true
447:     # Display menu template only if user preference is True
448:     if $session.pref_show_menus == true
449:       Tgtemplate::display(template, menudata)
450:     end
451: 
452:     # Display the prompt
453:     Tgtemplate::display('menu_prompt.ftl')
454:     selection = self.inputkey.upcase
455:     # Check if user supplied selection is included in validkeys array
456:     if validkeys.include?(selection)
457:       # Check for Global Input Values with prepended slash
458:        case selection
459:          when "/?" # Show the shortcut / commands menu
460:            Tgtemplate::display('menu_shortcuts.ftl')
461:          when "/G" # Fast Goodybye
462:            Telegard::goodbye_fast
463:            return 0 
464:          when "/H" # Help Menu - probably dont want this here
465:            Telegard.unimplemented
466:          when "/M" # Toggle $session.prefs_show_menus
467:            $session.pref_show_menus = $session.pref_show_menus.toggle
468:            $session.save
469:            Tgtemplate::display('toggle_show_menus.ftl', {"togglevalue" => $session.pref_show_menus.to_s})
470:          when "/P" # Toggle $session.prefs_term_pager
471:            $session.pref_term_pager = $session.pref_term_pager.toggle
472:            $session.save
473:            Tgtemplate::display('toggle_term_pager.ftl', {"togglevalue" => $session.pref_term_pager.toggle.to_s})
474:          when "/T" # Show time remaining
475:            Tgtemplate::display('time_remain.ftl', {
476:                    'timeremain'=> $session.timeremain.to_s,
477:                    'curtime' => Time.now.to_s,
478:                    'expires' => $session.expires.to_s
479:            })
480:          when "/W"
481:            puts "Who's online"
482:          when "/X"
483:            puts "Fast Return to Main"
484:        end
485:         done = true
486:     else
487:       Tgtemplate::display('menu_invalidkey.ftl')
488:     end
489:   end
490:   return selection
