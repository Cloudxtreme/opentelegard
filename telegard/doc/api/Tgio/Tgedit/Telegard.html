def edit
  loop = true
  banner
  while loop == true
    row = inputrow
    if row == 1
      loop = false
    elsif row == 2
      loop = false
      return nil
    end
  end #/while
  body = ""
  @buffer.each do |line|
    body = body + line + "\n"
  end
  return body
end

    # File bbs/lib/tgio/tgedit.rb, line 57
def inputrow
  skipline = false
  newline = Tgio::Input.inputeditline(79)
  if newline.length == 2
    case newline
      when "/A" # Abort
        Tgtemplate.display('tgedit_abort.ftl')
        return 2
      when "/X" # Exit and Save
        return 1
      when "/H" # Help
        Tgtemplate.display('tgedit_help.ftl')
        skipline = true
      when "/P" # Preview & Change
        preview
        skipline = true
    end #/case
  end #/if
  @buffer.push(newline) unless skipline == true
  return 0
end

def preview
  linenum = 0
  puts "\n"
  @buffer.each do |line|
    linenum +=1
    puts "#{linenum.to_s}:#{line}"
  end
end
